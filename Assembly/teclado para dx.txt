; Escrever um procedimento que permita carregar em Dx um número inteiro decimal a partir do teclado. O procedimento deve garantir que só sejam digitados números. 

data segment
    erro db "Caracter invalido, digite um numero.$"
    pkey db "press any key...$"
ends

stack segment
    dw   128  dup(0)
ends

code segment
start:
; set segment registers:
    mov ax, data
    mov ds, ax
    mov es, ax

leitura:
    mov ah,8  ;le sem eco
    int 21h
    
    cmp al, 48
    ja segundaCmp
    
    lea dx,erro
    mov ah, 9
    int 21h
         
    mov dl, 10
    mov ah,2    ;nova linha
    int 21h
    
    mov dl, 13
    mov ah,2    ;retorna o carro
    int 21h
    
    jmp leitura
              
              
              
segundaCmp:
    cmp al, 58
    jb ehNumero 
    
    lea dx,erro
    mov ah, 9
    int 21h
         
    mov dl, 10
    mov ah,2    ;nova linha
    int 21h
    
    mov dl, 13
    mov ah,2    ;retorna o carro
    int 21h
    
    jmp leitura
    
    
ehNumero:
    mov dl, al 
    
    mov ah, 4ch
    int 21h
    
    
    
            
    lea dx, pkey
    mov ah, 9
    int 21h        ; output string at ds:dx
    
    ; wait for any key....    
    mov ah, 1
    int 21h
    
    mov ax, 4c00h ; exit to operating system.
    int 21h    
ends

end start ; set entry point and stop the assembler.
